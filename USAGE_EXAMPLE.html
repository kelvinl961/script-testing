API Methods

// Check if available
console.log(window.PWAInstallBanner); // Should NOT be undefined

// Initialize
window.PWAInstallBanner.init({ lang: 'bn' });

// Show banner manually
window.PWAInstallBanner.show();

// Hide banner
window.PWAInstallBanner.hide();

// Get current language
console.log(window.PWAInstallBanner.getLang()); // 'en' or 'bn'

// Change language dynamically
window.PWAInstallBanner.setLang('bn');

// Check if PWA is installed
console.log(window.PWAInstallBanner.isStandalone());

// Get version
console.log(window.PWAInstallBanner.version); // '2.1.1'
